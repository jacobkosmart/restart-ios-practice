//  OnboardingView.swift

import SwiftUI

struct OnboardingView: View {
// MARK: PROPERTY
@AppStorage("onboading") var isOnboadingViewActive: Bool = true

// MARK: - BODY
var body: some View {
VStack(spacing: 20) {
Text("Onboarding").font(.largeTitle)

Button(action: {
	isOnboadingViewActive = false
}) {
	Text("Start")
}
}///: VStack
}
}
